(\n        <p style={{ textAlign: 'center', padding: '2rem' }}>Cargando...</p>\n      ) : (\n        <PolaroidGrid \n          peliculas={peliculas} \n          onSeleccionar={(item) => {\n            setItemSeleccionado(item);\n            setModalAbierto(true);\n          }}\n        />\n      )}\n      <DetailModal \n        item={itemSeleccionado}\n        abierto={modalAbierto}\n        onCerrar={() => setModalAbierto(false)}\n      />\n    </div>\n  );\n}\n```\n\n---\n\n## Ejemplo 2: SearchFilter.jsx\n\n```jsx\nimport React, { useState } from 'react';\n\nexport default function SearchFilter({ onBuscar }) {\n  const [consulta, setConsulta] = useState('');\n  const [ordenPor, setOrdenPor] = useState('popularity');\n  const [orden, setOrden] = useState('desc');\n\n  const manejarBuscar = (e) => {\n    e.preventDefault();\n    if (consulta.trim()) {\n      onBuscar(consulta.trim());\n      setConsulta('');\n    }\n  };\n\n  return (\n    <div className=\"search-filter-section\">\n      <form className=\"search-box\" onSubmit={manejarBuscar}>\n        <input \n          type=\"text\" \n          id=\"search-input\" \n          placeholder=\"Buscar película o serie...\"\n          value={consulta}\n          onChange={(e) => setConsulta(e.target.value)}\n        />\n        <button type=\"submit\" className=\"search-btn\">Buscar</button>\n      </form>\n      \n      <div className=\"filters-box\">\n        <div className=\"filter-group\">\n          <label htmlFor=\"order-filter\">Ordenar por:</label>\n          <select \n            id=\"order-filter\"\n            value={ordenPor}\n            onChange={(e) => setOrdenPor(e.target.value)}\n          >\n            <option value=\"popularity\">Popularidad</option>\n            <option value=\"rating\">Rating</option>\n            <option value=\"release_date\">Fecha de Salida</option>\n          </select>\n        </div>\n\n        <div className=\"filter-group\">\n          <label htmlFor=\"sort-filter\">Orden:</label>\n          <select \n            id=\"sort-filter\"\n            value={orden}\n            onChange={(e) => setOrden(e.target.value)}\n          >\n            <option value=\"desc\">Descendente</option>\n            <option value=\"asc\">Ascendente</option>\n          </select>\n        </div>\n      </div>\n    </div>\n  );\n}\n```\n\n---\n\n## Ejemplo 3: PolaroidGrid.jsx\n\n```jsx\nimport React from 'react';\nimport PolaroidCard from './PolaroidCard';\n\nexport default function {peliculas.map((item, index) => (\n          <PolaroidCard \n            key={item.id} \n            item={item}\n            onSeleccionar={onSeleccionar}\n            index={index}\n          />\n        ))}\n      </div>\n    </div>\n  );\n}\n```\n\n---\n\n## Ejemplo 4: PolaroidCard.jsx\n\n```jsx\nimport React from 'react';\n\nexport default function PolaroidCard({ item, onSeleccionar, index }) {\n  const titulo = item.title || item.name;\n  const rating = (item.vote_average / 2).toFixed(1);\n  const poster = item.poster_path \n    ? `https://image.tmdb.org/t/p/w500${item.poster_path}` \n    : 'https://via.placeholder.com/300x450?text=Sin+imagen';\n  const esRosa = index % 2 === 1;\n\n  // Determina el color según el rating\n  let colorClass = 'score-low';\n  if (item.vote_average >= 7) colorClass = 'score-high';\n  else if (item.vote_average >= 5) colorClass = 'score-medium';\n\n  return (\n    <div className=\"polaroid\" onClick={() => onSeleccionar(item)}>\n      <div className={`washi-tape ${esRosa ? 'pink-tape' : ''}`}></div>\n      <div className=\"photo-frame\">\n        <img src={poster} alt={titulo} />\n        <div className={`sticker-score ${colorClass}`}>{rating}</div>\n      </div>\n      <div className=\"handwritten-caption\">\n        <h3>{titulo}</h3>\n      </div>\n    </div>\n  );\n}\n```\n\n---\n\n## Ejemplo 5: DetailModal.jsx\n\n```jsx\nimport React, { useEffect, useRef } from 'react';\n\nexport default function DetailModal({ item, abierto, onCerrar }) {\n  const dialogRef = useRef(null);\n\n  useEffect(() => {\n    if (abierto && item && dialogRef.current) {\n      dialogRef.current.showModal();\n    } else if (!abierto && dialogRef.current) {\n      dialogRef.current.close();\n    }\n  }, [abierto, item]);\n\n  if (!item) return null;\n\n  const titulo = item.title || item.name;\n  const year = item.release_date \n    ? item.release_date.split('-')[0] \n    : item.first_air_date?.split('-')[0];\n  const fondo = item.backdrop_path \n    ? `https://image.tmdb.org/t/p/w500${item.backdrop_path}` \n    : 'https://via.placeholder.com/300x300';\n  const descripcion = item.overview || 'Sin descripción disponible';\n\n  return (\n    <dialog \n      ref={dialogRef} \n      className=\"modal-overlay\" \n      onClick={(e) => e.target === dialogRef.current && onCerrar()}\n    >\n      <button className=\"heart-close-btn\" onClick={onCerrar}>\n        <span>✖</span>\n      </button>\n\n      <div className=\"deco-header\">\n        <div className=\"header-pattern\"></div>\n        <div className=\"hero-circle\">\n          <img src={fondo} alt={titulo} />\n        </div>\n      </div>\n\n      <div className=\"deco-body\">\n        <div className=\"title-section\">\n          <h2 className=\"cute-title\">{titulo}</h2>\n          <div className=\"cute-badges\">\n            {year && <span className=\"badge badge-yellow badge-time\">{year}</span>}\n          </div>\n        </div>\n\n        <div className=\"bubble-desc\">\n          <p>{descripcion}</p>\n        </div>\n      </div>\n    </dialog>\n  );\n}\n```\n\n---\n\n## Ejemplo 6: Navigation.jsx (con funcionalidad)\n\n```jsx\nimport React from 'react';\n\nexport default function Navigation({ tipo, alCambiarTipo }) {\n  return (\n    <nav className=\"ribbon-nav\">\n      <button \n        className={`ribbon-btn ${tipo === 'movie' ?
